日本地図: 普通教室のエアコン設置率

文科省公立学校施設の空調（冷房）設備設置状況の結果についてから2017年4月1日現在の小中学校普通教室のエアコン設置率をグラフにした。

# install.packages("NipponMap")
library(NipponMap)
x = read.csv("https://okumuralab.org/~okumura/stat/data/aircon_shochu.csv", fileEncoding="UTF-8")
c = x[,4] / 100   # 範囲 [0,1]
cols = colorRamp(c("#ff2800","white","#0041ff"))
JapanPrefMap(col=rgb(cols(c)/255))
rect(142+(0:99)/20, 32, 142+(1:100)/20, 33, col=rgb(cols(0:99/99)/255), border=NA)
rect(142, 32, 147, 33)
text(142+(0:2)*2.5, 32, c("0%","50%","100%"), pos=1)
