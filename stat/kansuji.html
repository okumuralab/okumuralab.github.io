漢数字

平成22年度の入試センター試験問題「情報関係基礎」第3問をRで解いてみましょう。

kansuji = function(n) {
  Suji = c("","二","三","四","五","六","七","八","九")
  KuraiMoji = c("","十","百","千")
  kurai = 1000
  for (keta in 4:1) {
    d = n %/% kurai
    if (d != 0) {
      if (d == 1 && keta == 1) {
        cat("一")
      } else {
        cat(Suji[d])
        cat(KuraiMoji[keta])
      }
    }
    n = n %% kurai
    kurai = kurai %/% 10
  }
}

%/% は整数の割り算，%% は割り算の余り，cat(...) は出力を意味します。
